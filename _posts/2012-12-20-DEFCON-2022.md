```python
import zlib
from string import printable
from itertools import combinations

goal = hex(zlib.crc32(b'the'))
team = b'Arr3stY0u'

count = 0
for i in range(0, 15):
        print(f"trying {i} letters added: ")
        combs = combinations(printable[:62], i)
        for m in combs:
                m = team + "".join(m).encode()
                count += 1
                #print(m)
                hsh = hex(zlib.crc32(m))
                if hsh == goal:
                        print(m)
```
